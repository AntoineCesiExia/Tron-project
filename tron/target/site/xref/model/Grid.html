36  	 * 
37  	 * @param lightcycle
38  	 *            check collision between LightCycle
39  	 * 
40  	 * @return return if there is a collision
41  	 */
42  	public boolean checkCollision(LightCyclehref="../model/elements/LightCycle.html#LightCycle">LightCycle LightCycle) {
43  		if (LightCycle.getX() > this.nbrCellWidth || LightCycle.getX() <= 0 || LightCycle.getY() <= 0 || LightCycle.getY() > this.nbrCellHeight) {
44  			LightCycle.setAlive(false);
45  			return true;
46  		}
47  
48  		for (IElements LightWall : LightCycle1.getLightWall()) {
49  			if (LightCycle.getX() == LightWall.getX() && LightCycle.getY() == LightWall.getY()) {
50  				LightCycle.setAlive(false);
51  				return true;
52  			}
53  		}
54  
55  		for (IElements LightWall : LightCycle2.getLightWall()) {
56  			if (LightCycle.getX() == LightWall.getX() && LightCycle.getY() == LightWall.getY()) {
57  				LightCycle.setAlive(false);
58  				return true;
59  			}
60  		}
61  		
62  		return false;
63  	}
